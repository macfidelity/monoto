  4 /**
  5  * @name getJavaScriptVersions
  6  * @description gets the version numbers of the major JavaScript Libraries and displays them in admin view
  7  * @memberof admin
  8  */
  9 function getJavaScriptVersions()
 10 {
 11     console.debug("getJavaScriptVersions ::: Start");
 12 
 13     // BootStrap
 14     //
 15     // get bootstrap version
 16     var bootstrapVersion = ($().modal||$().tab).Constructor.VERSION.split(',');
 17     // update gui
 18     $("#libVersionBootstrap").val(bootstrapVersion);
 19 
 20     // ckeditor
 21     //
 22     // update gui
 23     $("#libVersionCKEditor").val(CKEDITOR.version);
 24 
 25     // DataTables
 26     //
 27     // get DataTables version
 28     var datatableVersion = $.fn.dataTable.version;
 29     // update gui
 30     $("#libVersionDataTable").val(datatableVersion);
 31 
 32     // jquery
 33     //
 34     if (typeof jQuery != 'undefined')
 35     {
 36         // update gui
 37         $("#libVersionJQuery").val(jQuery.fn.jquery);
 38     }
 39 
 40     console.debug("getJavaScriptVersions ::: Stop");
 41 }
