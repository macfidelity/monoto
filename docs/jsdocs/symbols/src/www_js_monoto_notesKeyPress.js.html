1 /** @namespace */
  2  var notesKeyHandler = {};
  3 
  4 document.onkeyup = KeyCheck;
  5 
  6 
  7 /**
  8  * @description general key-press handler for notes
  9  * @memberof notesKeyHandler
 10  * @param {event} e - Event / Key
 11  */
 12 function KeyCheck(e)
 13 {
 14     var KeyID = (window.event) ? event.keyCode : e.keyCode;
 15 
 16     console.log("KeyCheck ::: Key press: " + KeyID);
 17 
 18     switch(KeyID)
 19     {
 20         // ESC - reset the UI
 21         case 27:
 22             resetNotesUI();
 23             break;
 24 
 25         // Arrow Up
 26         case 38:
 27             // specialcase: if focus in search -> jump to first record in table
 28             if(document.activeElement.id === "searchField")
 29             {
 30                 selectUpperRow();
 31             }
 32             break;
 33 
 34         // Arrow Down
 35         case 40:
 36             // specialcase: if focus in search -> jump to first record in table
 37             if(document.activeElement.id === "searchField")
 38             {
 39                 selectNextRow();
 40             }
 41             break;
 42 
 43         // DEL - delete selected note & reloads page
 44         case 46:
 45             var deleteID = $("#noteID").val();
 46             if(deleteID !== "")
 47             {
 48                 deleteNote(); // execute delete function
 49             }
 50             break;
 51 
 52 
 53         // F2 - trigger maximize editor to fullscreen
 54         case 113:
 55             var noteID = $("#noteID").val();
 56             if(noteID !== "")
 57             {
 58                 CKEDITOR.instances.editor1.execCommand( "maximize" );
 59             }
 60             else
 61             {
 62                 console.log("nothing to mazimize - as no note is selected");
 63             }
 64             break;
 65 
 66         // F5 - Reload main page
 67         case 116:
 68             reloadCurrentPage(); // execute reload function
 69             break;
 70 
 71         // F9 - save
 72         case 120:
 73             saveNote();
 74             break;
 75 
 76     } // end switch
 77 } // end function keycheck
