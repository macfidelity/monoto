 25 /**
 26  * @description realizes a password change
 27  * @memberof profile
 28  */
 29 function doChangeProfilePassword()
 30 {
 31     console.debug("doChangeProfilePassword ::: Start");
 32 
 33     var password = $("#newPassword").val();
 34 
 35     var jqxhr = $.post( "inc/profileChangeUserPW.php", { password: password }, function()
 36     {
 37         console.log("doChangeProfilePassword ::: successfully updated user password");
 38     })
 39     .done(function()
 40     {
 41         console.log("doChangeProfilePassword ::: done");
 42         createNoty("Successfully changed password","success");
 43 
 44         // reset password fields
 45         $("#newPassword").val("");
 46         $("#newPasswordConfirm").val("");
 47 
 48         // disable button
 49         $("#bt_continue").prop("disabled",true);
 50     })
 51     .fail(function(jqxhr, textStatus, errorThrown)
 52     {
 53         console.error("FAIL");
 54         console.log(jqxhr);
 55         console.log(textStatus);
 56         console.log(errorThrown);
 57 
 58         alert("Updating password failed.");
 59     })
 60     .always(function()
 61     {
 62         // doing nothing so far
 63     });
 64 
 65     console.debug("doChangeProfilePassword ::: Stop");
 66 }

107 /**
108  * @description deletes all events from the current user account
109  * @memberof profile
110  */
111 function deleteAllMyUserEvents()
112 {
113     console.debug("deleteAllMyUserEvents ::: Start Delete-All-My-User-Events-Dialog.");
114 
115     var x = noty({
116         text: "Really delete all your events from log?",
117         type: "confirm",
118         dismissQueue: false,
119         layout: "topRight",
120         theme: "defaultTheme",
121         buttons: [
122             {
123                 addClass: "btn btn-primary", text: "Ok", onClick: function($noty)
124                 {
125                     $noty.close();
126                     $.post("inc/profileDeleteMyUserEvents.php");
127                     createNoty("Deleted all events from log","success");
128                     location.reload();
129                 }
130             },
131             {
132                 addClass: "btn btn-danger", text: "Cancel", onClick: function($noty)
133                 {
134                     $noty.close();
135                     createNoty("Aborted","information");
136                 }
137             }
138         ]
139     });
140 
141     console.debug("deleteAllMyUserEvents ::: Finished Delete-All-My-User-Events-Dialog.");
142 }
