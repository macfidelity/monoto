 22 /**
 23  * @description Calculates password strength
 24  * @memberof generic
 25  */
 26 function passwordStrength()
 27 {
 28     console.debug("passwordStrength ::: Start");
 29 
 30     var strongRegex = new RegExp("^(?=.{10,})(?=.*[A-Z])(?=.*[a-z])(?=.*[0-9])(?=.*\\W).*$", "g");
 31     var mediumRegex = new RegExp("^(?=.{9,})(((?=.*[A-Z])(?=.*[a-z]))|((?=.*[A-Z])(?=.*[0-9]))|((?=.*[a-z])(?=.*[0-9]))).*$", "g");
 32     var enoughRegex = new RegExp("(?=.{8,}).*", "g");
 33 
 34     curPasswordString = $("#newPassword").val();
 35 
 36     if (false === enoughRegex.test(curPasswordString)) // < 8 chars
 37     {
 38         console.log("passwordStrength ::: More characters");
 39         $("#passstrength").html("More Characters");
 40     }
 41     else if (strongRegex.test(curPasswordString))
 42     {
 43         console.log("passwordStrength ::: OK");
 44         $("#passstrength").className = "ok";
 45         $("#passstrength").html("Strong");
 46      }
 47      else if (mediumRegex.test(curPasswordString))
 48      {
 49         console.log("passwordStrength ::: alert");
 50         $("#passstrength").className = "alert";
 51         $("#passstrength").html("Medium");
 52      }
 53      else
 54      {
 55         console.log("passwordStrength ::: error");
 56         $("#passstrength").className = "error";
 57         $("#passstrength").html("Weak");
 58      }
 59 
 60     console.debug("passwordStrength ::: End");
 61 
 62     //return true;
 63 }

 66 /**
 67  * @description Compare password change fields and enable or disable the update button
 68  * @memberof generic
 69  */
 70 function validatePasswordChangeInput()
 71 {
 72     console.debug("validatePasswordChangeInput ::: Start");
 73 
 74     if ($("#newPassword").val() === $("#newPasswordConfirm").val())
 75     {
 76             // password & passwordConfirm do match
 77 
 78             // now check if min length is reached
 79         if($("#newPassword").val().length > 7)
 80         {
 81                 console.log("p.php ::: Passwords do match and min length is reached");
 82                 // enable button
 83                 $("#bt_continue").prop("disabled",false);
 84                 // update status icon
 85                 $("#passwordDiff").text("");
 86                 $("#passwordDiff").append("<i class='fas fa-smile'></i>");
 87         }
 88         else
 89         {
 90             console.warn("p.php ::: Passwords do match but min length is not reached");
 91             // disable button
 92             $("#bt_continue").prop("disabled",true);
 93             // update status icon
 94             $("#passwordDiff").text("");
 95             $("#passwordDiff").append("<i class='far fa-frown'></i>");
 96         }
 97     }
 98     else
 99     {
100         console.warn("p.php ::: Passwords do not match");
101         // disable button
102         $("#bt_continue").prop("disabled",true);
103         // update status icon
104         $("#passwordDiff").text("");
105         $("#passwordDiff").append("<i class='far fa-frown'></i>");
106     }
107 
108     console.debug("validatePasswordChangeInput ::: End");
109 }
